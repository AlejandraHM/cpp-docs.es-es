'section' : solamente se permite si se anida directamente bajo una directiva 'sections' de OpenMP

```  
// C3044.cpp  
// compile with: /openmp /c  
#include "omp.h"  
int main() {  
   int n2 = 2, n3 = 3;  
  
   #pragma omp parallel  
   {  
      ++n2;  
  
      #pragma omp sections

         }  
      }  
   }  
}
```
