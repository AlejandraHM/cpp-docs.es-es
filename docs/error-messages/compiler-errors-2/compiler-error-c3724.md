# <a name="compiler-error-c3724"></a>Error del compilador C3724
debe #include \<windows.h > usar multithreading con eventos  
  
 El archivo de windows.h es necesario si utiliza multithreading con eventos. Para corregir este error, agregue `#include <windows.h>` a la parte superior del archivo en las que los or√≠genes de eventos y los eventos se definen los receptores.  
  
```  
// C3724.cpp  
// uncomment the following line to resolve  
// #include <windows.h>  
  
[event_source(native), threading(apartment)]  
class CEventSrc {  
public:  
   __event void event1();   // C3724  
};  
  
[event_receiver(native)]  
class CEventRec {  
public:  
   void handler1() {  
   }  
  
   void HookEvents(CEventSrc* pSrc) {  
      __hook(CEventSrc::event1, pSrc, CEventRec::handler1);  
   }  
  
   void UnhookEvents(CEventSrc* pSrc) {  
      __unhook(CEventSrc::event1, pSrc, CEventRec::handler1);  
   }  
};  
  
int main() {  
}  
```
