# Error del compilador C3724
[!INCLUDE[vs2017banner](../../assembler/inline/includes/vs2017banner.md)]

\#include \<windows.h de la necesidad\> de utilizar el multi\- subprocesamiento con eventos  
  
 Se requiere el archivo windows.h si se utiliza el subprocesamiento múltiple en eventos.  Para resolver este error, agregue `#include <windows.h>` al principio del archivo donde se definen los orígenes y receptores de eventos.  
  
```  
// C3724.cpp  
// uncomment the following line to resolve  
// #include <windows.h>  
  
[event_source(native), threading(apartment)]  
class CEventSrc {  
public:  
   __event void event1();   // C3724  
};  
  
[event_receiver(native)]  
class CEventRec {  
public:  
   void handler1() {  
   }  
  
   void HookEvents(CEventSrc* pSrc) {  
      __hook(CEventSrc::event1, pSrc, CEventRec::handler1);  
   }  
  
   void UnhookEvents(CEventSrc* pSrc) {  
      __unhook(CEventSrc::event1, pSrc, CEventRec::handler1);  
   }  
};  
  
int main() {  
}  
```
